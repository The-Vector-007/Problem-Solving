int countLeafNodes( TreeNode<int> * root ){
    if( root == NULL ){
        return 0;
    }

    if( root->children.size() == 0 ){
        return 1;
    }

    int count = 0;

    for( int i = 0; i < root->children.size(); i++ ){
        count += countLeafNodes( root->children[i] );
    }

    return count;
}
